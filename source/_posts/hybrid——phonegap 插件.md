# 命令
```
#  查看插件 
cordova plugin list
cordova plugins



#  删除插件
cordova plugin remove com.seed.cordova.huhu



#  添加插件
cordova plugin add  org.apache.cordova.file



# 添加本地插件 
cordova plugin add C:\**\Plugins\cordova-plugin-huhu



```
